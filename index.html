              (defn- ellipsify [node text]
                (u/set-node-text! node text)
                (let [height (-> node .-clientHeight)]
                  (when-not (node-within-height node height)
                    (loop [viewable-words []
                           rest-of-words (str/split text #"\s+")
                           text ""]
                      (let [last-word? (= (count rest-of-words) 1)
                            viewable-words (conj viewable-words (first rest-of-words))
                            new-text (str (str/join " " viewable-words) (if (not last-word?) "..." ""))]
                        (u/set-node-text! node new-text)
                        (let [text-overflows? (not (node-within-height node height))]
                          (if (or last-word? text-overflows?)
                            (when text-overflows?
                              (u/set-node-text! node text))
                            (recur viewable-words 
                                   (drop 1 rest-of-words)
                                   new-text))))))))
